% This Latex header breaks proper indenting of headings smaller
% than second level in the table of contents when --toc is used.

\usepackage{sectsty}

\linespread{1.3}

\newfontfamily\sfprodisplaybold{SF Pro Display Bold}
\newfontfamily\sfprodisplayregularitalic{SF Pro Display Regular Italic}

\makeatletter
  \renewcommand\section{\@startsection {section}{1}{\z@}%
  {-3.5ex \@plus -1ex \@minus -0.2ex}%
  {1.0ex \@plus0.2ex}%
  {\fontsize{16pt}{16pt}\selectfont\sfprodisplaybold}}
\makeatother

\makeatletter
  \renewcommand\subsection{\@startsection {subsection}{1}{\z@}%
  {-3.5ex \@plus -1ex \@minus -0.2ex}%
  {0.5ex \@plus0.2ex}%
  {\fontsize{12pt}{12pt}\selectfont\sfprodisplaybold}}
\makeatother

\makeatletter
  \renewcommand\subsubsection{\@startsection {subsubsection}{1}{\z@}%
  {-3.5ex \@plus -1ex \@minus -0.2ex}%
  {0.5ex \@plus0.2ex}%
  {\fontsize{10pt}{10pt}\selectfont\sfprodisplaybold}}
\makeatother

\makeatletter
  \renewcommand\paragraph{\@startsection {paragraph}{1}{\z@}%
  {-3.5ex \@plus -1ex \@minus -0.2ex}%
  {0.5ex \@plus0.2ex}%
  {\fontsize{10pt}{10pt}\selectfont\sfprodisplayregularitalic}}
\makeatother
